create table EXPLICIT_IS_BEST
 ( x number, y char(100));
insert into EXPLICIT_IS_BEST
 select rownum, ‘padding’
 from all_objects
 where rownum < 10001;
create or replace
 procedure implicit is
 dummy number;
 begin
 for i in 1 .. 50000 loop
 select 1
 into dummy
 from explicit_is_best
 where x = 1;
 end loop;
end;
/
create or replace
 procedure explicit is
 cursor explicit_cur is
 select 1
 from explicit_is_best
 where x = 1;
 dummy number;
 begin
 for i in 1 .. 50000 loop
 open explicit_cur;
 fetch explicit_cur
 into dummy;
 close explicit_cur;
 end loop;
 end;
/
